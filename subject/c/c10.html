                #include <stdio.h>

                #define SWAP(x, y) do { \
                    typeof(x) _tmp = x; \
                    x = y; \
                    y = _tmp; \
                } while(0)
                
                int main() {
                    int a = 10, b = 20;
                    printf("Before swap: a=%d, b=%d\n", a, b);
                    SWAP(a, b);
                    printf("After swap: a=%d, b=%d\n", a, b);
                    return 0;
                }

Output:

          
            Before swap: a=10, b=20
            After swap: a=20, b=10

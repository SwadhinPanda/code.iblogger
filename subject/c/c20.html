Download .txt 
            
                        #include <stdio.h>
                       #include <stdlib.h>
                    
                        int main() {
                        FILE *source_file, *target_file;
                        char source_name[100], target_name[100];
                        char c;
                    
                        printf("Enter the name of the source file: ");
                        scanf("%s", source_name);
                    
                        source_file = fopen(source_name, "r");
                    
                        if (source_file == NULL) {
                            printf("Unable to open source file!\n");
                            exit(1);
                        }
                    
                        printf("Enter the name of the target file: ");
                        scanf("%s", target_name);
                    
                        target_file = fopen(target_name, "w");
                    
                        if (target_file == NULL) {
                            printf("Unable to create target file!\n");
                            fclose(source_file);
                            exit(1);
                        }
                    
                        while ((c = fgetc(source_file)) != EOF) {
                            fputc(c, target_file);
                        }
                    
                        printf("File copied successfully.\n");
                    
                        fclose(source_file);
                        fclose(target_file);
                    
                        return 0;
                       }

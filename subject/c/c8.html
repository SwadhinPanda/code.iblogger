                #include <stdio.h>

                int isPrime(int num) {
                    if (num <= 1) {
                        return 0; 
                    }
                    for (int i = 2; i <= num/2; i++) {
                        if (num % i == 0) {
                            return 0;  
                        }
                    }
                    return 1;  
                }
                
                int main() {
                    printf("Prime numbers less than 100: ");
                    for (int i = 2; i < 100; i++) {
                        if (isPrime(i)) {
                            printf("%d ", i);
                        }
                    }
                
                 return 0;
                }
